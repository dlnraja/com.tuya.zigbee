FLOW CARD STRUCTURE:
driver.flow.compose.json format:
{
  "triggers": [
    {
      "id": "button_pressed",
      "title": { "en": "Button pressed", "fr": "Bouton appuyé" },
      "tokens": [
        { "name": "button", "type": "string", "title": { "en": "Button" } },
        { "name": "action", "type": "string", "title": { "en": "Action" } }
      ],
      "args": [
        { "name": "device", "type": "device", "filter": "driver_id=xxx" }
      ]
    }
  ],
  "actions": [...],
  "conditions": [...]
}

IMPLEMENTATION PATTERN:
// driver.js
this.buttonPressedTrigger = this.homey.flow.getDeviceTriggerCard('button_pressed');

// device.js
async onButton(button, action) {
  await this.driver.buttonPressedTrigger
    .trigger(this, { button: String(button), action: String(action) })
    .catch(this.error);
}
