SDK3 Implementation:
async setupIASZone() {
  // 1. Write CIE Address (SDK3 method)
  await endpoint.clusters[1280].writeAttributes({
    iasCIEAddress: this.homey.zigbee.ieee
  });
  
  // 2. Listen for zone status changes
  endpoint.clusters[1280].on('zoneStatusChangeNotification', (notif) => {
    const alarm = notif.zoneStatus.alarm1 === 1;
    this.setCapabilityValue('alarm_motion', alarm);
  });
  
  // 3. Setup enrollment handler
  endpoint.clusters[1280].onZoneEnrollRequest = async () => {
    await endpoint.clusters[1280].zoneEnrollResponse({
      enrollResponseCode: 0,
      zoneId: 10
    });
  };
  
  // 4. Proactive enrollment (SDK3 best practice)
  await endpoint.clusters[1280].zoneEnrollResponse({
    enrollResponseCode: 0,
    zoneId: 10
  }).catch(() => {});
}
